#### loadVideo方法
- [x] 使用VideoFileService获取视频路径

**验证**:
```typescript
loadVideo() {
  const video = this.videoList[this.currentVideoIndex]
  const videoFileName = `第${video.id}课${video.subtitle}.mp4`
  
  // ✅ 已修改：使用外部视频文件路径
  if (this.videoFileService) {
    this.videoSrc = this.videoFileService.getVideoPath(videoFileName)
    
    // ✅ 已添加：检查视频文件是否存在
    if (!this.videoFileService.checkVideoExists(videoFileName)) {
      console.error(`❌ 视频文件不存在: ${videoFileName}`)
      console.error(`请确保视频文件已放置在: ${this.videoFileService.getExternalVideoDir()}`)
    }
  } else {
    console.error('❌ VideoFileService 未初始化')
  }
  
  console.info(`📹 加载视频: ${this.videoSrc}`)
}
```
