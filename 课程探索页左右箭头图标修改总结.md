#### 代码修改
```typescript
// 修改前（文本）
Stack() {
  Column()
    .width(80)
    .height(80)
    .borderRadius(40)
    .linearGradient({
      angle: 135,
      colors: [['#3b82f6', 0], ['#1d4ed8', 1]]
    })
    .shadow({ 
      radius: 30, 
      color: 'rgba(59, 130, 246, 0.6)', 
      offsetX: 0,
      offsetY: 12
    })
    .opacity(this.currentPage === 0 ? 0.3 : 1)

  Text('◀')
    .fontSize(48)
    .fontColor(Color.White)
}

// 修改后（图片）
Image($r('app.media.ic_arrow_left'))
  .width(80)
  .height(80)
  .opacity(this.currentPage === 0 ? 0.3 : 1)
  .shadow({ 
    radius: 30, 
    color: 'rgba(59, 130, 246, 0.6)',  // 蓝色阴影
    offsetX: 0,
    offsetY: 12
  })
  .onClick(() => {
    this.previousPage()
  })
```
