## 🔧 技术实现

### 页面结构
```typescript
@Entry
@Component
struct ProfilePage {
  @State currentPageIndex: number = 0
  @State userName: string = '小小鱼'
  @State antiAddictionTime: number = 30
  @State showTimeDialog: boolean = false
  @State inputTime: string = '30'
  private swiperController: SwiperController = new SwiperController()
  private dataPreferences: preferences.Preferences | null = null
}
```

### Swiper配置
```typescript
Swiper(this.swiperController) {
  this.ProfileInfoPage()    // 第1页：个人信息
  this.MenuPage()           // 第2页：功能菜单
}
.width('100%')
.layoutWeight(1)
.indicator(false)
.loop(false)
.duration(300)
.curve(Curve.EaseInOut)
.onChange((index: number) => {
  this.currentPageIndex = index
})
```

### 数据持久化
```typescript
// 初始化存储
async initPreferences() {
  const context = getContext(this) as common.UIAbilityContext
  this.dataPreferences = await preferences.getPreferences(context, 'profile_data')
}

// 加载防沉迷时间
async loadAntiAddictionTime() {
  this.antiAddictionTime = await this.dataPreferences.get('antiAddictionTime', 30)
}

// 保存防沉迷时间
async saveAntiAddictionTime(time: number) {
  await this.dataPreferences.put('antiAddictionTime', time)
  await this.dataPreferences.flush()
  promptAction.showToast({ message: '设置成功', duration: 2000 })
}
```

### 对话框实现
```typescript
@Builder
TimeSettingDialog() {
  Stack() {
    // 遮罩层
    Column()
      .width('100%')
      .height('100%')
      .backgroundColor('rgba(0, 0, 0, 0.5)')
      .onClick(() => {
        this.showTimeDialog = false
      })

    // 对话框内容
    Column({ space: 24 }) {
      Text('防沉迷设置')
      Text('请设置使用时长（分钟）')
      TextInput({ text: this.inputTime })
      Row({ space: 12 }) {
        Button('取消')
        Button('确定')
      }
    }
  }
  .zIndex(1000)
}
```
